-- assignment gets detected
find_rule = function(ast, rule)
   if(ast.rule == rule) then
      return ast
   end

   for i, n in ipairs(ast) do
      if(n.rule == rule) then
         return n
      elseif(n.rule) then
         local r = find_rule(n, rule)
         if(r) then
            return r
         end
      end
   end
end
